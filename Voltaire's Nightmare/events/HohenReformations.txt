country_event = {
	id = hohenref_events.1
	title = "hohenref_events.1.t"
	desc = "hohenref_events.1.d"
	picture = RELIGION_eventPicture
	
	major = yes
	fire_only_once = yes
	
	trigger = {
		religion = catholic
		NOT = { government = papal_government }
		reform_desire = 1.0
		NOT = { current_age = age_of_empire }
		is_year = 1100
		NOT = { has_global_flag = cathar_heresy_g }
		OR = {
			capital_scope = { superregion = french_superregion }
			NOT = { any_country = { capital_scope = { superregion = french_superregion } } }
		}
		capital_scope = {
			continent = europe
		}
		any_owned_province = {
			can_have_center_of_reformation_trigger = {
				RELIGION = cathar
			}
		}	
	}

	immediate = {
		hidden_effect = {
			set_global_flag = cathar_heresy_g
			enable_religion = cathar
		}
	}
	
	mean_time_to_happen = {
		months = 500
		
		modifier = {
			factor = 0.1
			NOT = {
				calc_true_if = {
					all_country = { capital_scope = { superregion = french_superregion } }
					amount = 20
				}
			}
		}
		modifier = {
			factor = 0.9
			reform_desire = 1.10
		}
		modifier = {
			factor = 0.9
			reform_desire = 1.15
		}
		modifier = {
			factor = 0.9
			reform_desire = 1.20
		}
		modifier = {
			factor = 0.9
			reform_desire = 1.25
		}
		modifier = {
			factor = 0.9
			reform_desire = 1.30
		}
		modifier = {
			factor = 0.9
			NOT = { num_of_cities = 20 }
		}
		modifier = {
			factor = 1.25
			has_idea_group = religious_ideas
		}
		modifier = {
			factor = 0.95
			NOT = { has_opinion = { who = PAP value = -50 } }
		}
		modifier = {
			factor = 0.95
			NOT = { has_opinion = { who = PAP value = -100 } }
		}
		modifier = {
			factor = 0.95
			NOT = { has_opinion = { who = PAP value = -150 } }
		}		
		modifier = {
			factor = 0.85
			num_of_rebel_controlled_provinces = 1
		}
		modifier = {
			factor = 2.0
			OR = {
				government = theocratic_government
				government = monastic_order_government
			}
		}
		modifier = {
			factor = 4.0
			NOT = { is_year = 1135 }
		}
		modifier = {
			factor = 8.0
			OR = {
				culture_group = latin
				culture_group = rhaetic
				culture_group = basquegroup
				culture_group = iberian
			}
		}
	}

	option = {
		name = "hohenref_events.1.b"
		random_owned_province = {
			limit = {
				can_have_center_of_reformation_trigger = {
					RELIGION = cathar
				}
			}
			change_religion = cathar
			add_reform_center = cathar
			add_permanent_province_modifier = {
				name = "religious_zeal_at_conv"
				duration = 9000
			}
		}
		capital_scope = { change_religion = cathar }
		add_papal_influence = 10
		add_reform_desire = -2.00
		set_country_flag = cathar_reformation
	}
}
